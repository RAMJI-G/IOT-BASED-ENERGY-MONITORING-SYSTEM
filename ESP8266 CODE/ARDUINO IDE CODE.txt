// ================= BLYNK =================

#define BLYNK_TEMPLATE_ID "TMPL3dtdxhM1I"
#define BLYNK_TEMPLATE_NAME "IOT EMS SYSTEM"
#define BLYNK_AUTH_TOKEN "X4b1BKE2L6tNcIj_ZHIF35XbFlvCvs42"

#define BLYNK_PRINT Serial

#include <ESP8266WiFi.h>
#include <BlynkSimpleEsp8266.h>

// ================= OLED =================

#include <Wire.h>
#include <Adafruit_GFX.h>
#include <Adafruit_SSD1306.h>

// ================= WIFI =================

char ssid[] = "minato";
char pass[] = "";

// ================= OLED =================

#define SCREEN_WIDTH 128
#define SCREEN_HEIGHT 64

Adafruit_SSD1306 display(SCREEN_WIDTH, SCREEN_HEIGHT, &Wire, -1);

// ================= PINS =================

#define SENSOR_PIN A0
#define BUZZER D5

// ================= VALUES =================

float voltage = 230.0;
float current = 0.00;
float power   = 0.00;

float lastVoltage = 230.0;
float lastCurrent = 0.00;

String mode = "WAITING";

// ================= FILTER =================

float smoothVoltage = 230;
float smoothCurrent = 0;

// ================= TIMER =================

BlynkTimer timer;

// ======================================================
// STARTUP ANIMATION
// ======================================================

void startupAnimation()
{
  // SCREEN 1
  display.clearDisplay();

  display.setTextSize(3);

  display.setTextColor(WHITE);

  display.setCursor(8, 5);

  display.println("SMART");

  display.setTextSize(2);

  display.setCursor(38, 38);

  display.println("EMS");

  display.display();

  delay(1800);

  // SCREEN 2
  display.clearDisplay();

  display.setTextSize(1);

  display.setCursor(15, 8);

  display.println("ENERGY MONITORING");

  display.setCursor(42, 22);

  display.println("SYSTEM");

  display.drawRoundRect(10, 42, 108, 12, 3, WHITE);

  display.display();

  // LOADING BAR
  for(int i = 0; i <= 100; i += 4)
  {
    display.fillRoundRect(12, 44, i, 8, 2, WHITE);

    display.display();

    delay(60);
  }

  delay(500);

  // SCREEN 3
  display.clearDisplay();

  display.setTextSize(2);

  display.setCursor(15, 12);

  display.println("WELCOME");

  display.setTextSize(1);

  display.setCursor(10, 42);

  display.println("IoT ENERGY SYSTEM");

  display.display();

  delay(1800);

  // SCREEN 4
  display.clearDisplay();

  display.setTextSize(2);

  display.setCursor(10, 20);

  display.println("SYSTEM");

  display.setCursor(22, 42);

  display.println("READY");

  display.display();

  delay(1500);
}

// ======================================================
// STARTUP BUZZER
// ======================================================

void startupBeep()
{
  tone(BUZZER, 1200);

  delay(1000);

  noTone(BUZZER);
}

// ======================================================
// SEND DATA TO BLYNK
// ======================================================

void sendToBlynk()
{
  Blynk.virtualWrite(V0, voltage);

  Blynk.virtualWrite(V1, current);

  Blynk.virtualWrite(V2, power);
}

// ======================================================
// SMART FILTER
// ======================================================

float filteredADC()
{
  long total = 0;

  for(int i = 0; i < 50; i++)
  {
    total += analogRead(SENSOR_PIN);

    delay(1);
  }

  return total / 50.0;
}

// ======================================================
// SETUP
// ======================================================

void setup()
{
  Serial.begin(115200);

  pinMode(BUZZER, OUTPUT);

  // OLED
  Wire.begin(D2, D1);

  if(!display.begin(SSD1306_SWITCHCAPVCC, 0x3C))
  {
    while(1);
  }

  // INTRO
  startupAnimation();

  // BUZZER
  startupBeep();

  // WIFI + BLYNK
  Blynk.begin(BLYNK_AUTH_TOKEN, ssid, pass);

  // TIMER
  timer.setInterval(1000L, sendToBlynk);
}

// ======================================================
// LOOP
// ======================================================

void loop()
{
  Blynk.run();

  timer.run();

  // =====================================
  // READ ANALOG
  // =====================================

  float adc = filteredADC();

  float sensorVoltage =
  adc * (3.3 / 1023.0);

  // =====================================
  // MODE DETECTION
  // =====================================

  if(sensorVoltage > 1.7)
  {
    // =================================
    // CURRENT MODE
    // =================================

    mode = "CURRENT";

    // CALCULATION
    float rawCurrent =
    abs((sensorVoltage - 2.5) / 8.0);

    // REMOVE NOISE
    if(rawCurrent < 0.01)
    {
      rawCurrent = 0;
    }

    // SMOOTH FILTER
    smoothCurrent =
    (smoothCurrent * 0.85) +
    (rawCurrent * 0.15);

    current = smoothCurrent;

    // STORE
    lastCurrent = current;
  }

  else
  {
    // =================================
    // VOLTAGE MODE
    // =================================

    mode = "VOLTAGE";

    // ================================
    // FIXED VOLTAGE CALIBRATION
    // ================================

  // =================================
// FIXED VOLTAGE CALIBRATION
// =================================

float rawVoltage =
sensorVoltage * 18000;

    // LIMIT
    if(rawVoltage > 260)
    {
      rawVoltage = 260;
    }

    // SMOOTH FILTER
    smoothVoltage =
    (smoothVoltage * 0.85) +
    (rawVoltage * 0.15);

    voltage = smoothVoltage;

    // STORE
    lastVoltage = voltage;
  }

  // =====================================
  // USE STORED VALUES
  // =====================================

  voltage = lastVoltage;

  current = lastCurrent;

  // =====================================
  // SMART POWER CALCULATION
  // =====================================

  power = voltage * current;

  // NOISE CLEAN
  if(power < 0.5)
  {
    power = 0;
  }

  // =====================================
  // OLED UI
  // =====================================

  display.clearDisplay();

  display.setTextSize(1);

  display.setTextColor(WHITE);

  // TITLE
  display.setCursor(25, 0);

  display.println("SMART EMS");

  // LINE
  display.drawLine(0, 12, 128, 12, WHITE);

  // MODE
  display.setCursor(0, 16);

  display.print("MODE: ");

  display.println(mode);

  // VOLTAGE
  display.setCursor(0, 30);

  display.print("VOLTAGE: ");

  display.print(voltage, 1);

  display.println("V");

  // CURRENT
  display.setCursor(0, 42);

  display.print("CURRENT: ");

  display.print(current, 2);

  display.println("A");

  // POWER
  display.setCursor(0, 54);

  display.print("POWER  : ");

  display.print(power, 1);

  display.println("W");

  display.display();

  // =====================================
  // SERIAL
  // =====================================

  Serial.print("MODE: ");

  Serial.println(mode);

  Serial.print("Voltage: ");

  Serial.print(voltage);

  Serial.println("V");

  Serial.print("Current: ");

  Serial.print(current);

  Serial.println("A");

  Serial.print("Power: ");

  Serial.print(power);

  Serial.println("W");

  Serial.println("----------------");

  delay(200);
}
